``` bash?linenums

echo "lqdai" | sudo -S apt install -y expect

/usr/bin/expect << EOF
set timeout 5
spawn sudo passwd root
expect {
                  -ex {[sudo] password for lqdai:} {send "lqdai\r"}
				  -re {Enter new UNIX password:} {send "lqdai\r"}
				  -re {Retype new UNIX password:} {send "lqdai\r"}
}
expect eof
EOF

/usr/bin/expect << EOF

set timeout 30
spawn su root
expect {
                  "Password:" {send "lqdai\r"}
}
chmod u+w /etc/sudoers
echo "lqdai ALL=(ALL) NOPASSWD: ALL" >> /etc/sudoers
chmod u-w /etc/sudoers
EOF
```
